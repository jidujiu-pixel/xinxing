        function initProjects() {
            renderProjectCategories();

            // 添加项目标签点击事件
            document.querySelectorAll('.project-tab').forEach(tab => {
                tab.addEventListener('click', function() {
                    // 更新活动标签
                    document.querySelectorAll('.project-tab').forEach(t => t.classList.remove('active'));
                    this.classList.add('active');
                    
                    // 更新当前项目
                    currentProject = this.getAttribute('data-project');
                    currentCategory = 'all';
                    
                    // 重新渲染分类和图片
                    renderProjectCategories();
                    renderProjectImages();
                });
            });
        }

        // 渲染项目分类
        function renderProjectCategories() {
            const categoriesContainer = document.getElementById('project-categories');
            const project = projectsData[currentProject];
            
            let categoriesHTML = '<div class="project-category active" data-category="all">全部</div>';
            
            for (const category in project.categories) {
                categoriesHTML += `<div class="project-category" data-category="${category}">${category}</div>`;
            }
            
            categoriesContainer.innerHTML = categoriesHTML;
            
            // 添加分类点击事件
            document.querySelectorAll('.project-category').forEach(cat => {
                cat.addEventListener('click', function() {
                    // 更新活动分类
                    document.querySelectorAll('.project-category').forEach(c => c.classList.remove('active'));
                    this.classList.add('active');
                    
                    // 更新当前分类
                    currentCategory = this.getAttribute('data-category');
                    
                    // 重新渲染图片
                    renderProjectImages();
                });
            });
        }

        // 渲染项目图片
        function renderProjectImages() {
            const projectsContainer = document.getElementById('projects-grid');
            const project = projectsData[currentProject];
            
            let projectsHTML = '';
            let imageCount = 0;
            
            // 根据当前分类确定要显示的图片数量
            if (currentCategory === 'all') {
                // 显示所有分类的图片
                for (const category in project.categories) {
                    const count = project.categories[category];
                    for (let i = 1; i <= count; i++) {
                        projectsHTML += `

                }

            projectsContainer.innerHTML = projectsHTML;

        // 数字动画
        function animateNumbers() {
            const statNumbers = document.querySelectorAll('.stat-number');
            statNumbers.forEach(stat => {
                const target = parseInt(stat.getAttribute('data-count'));
                const duration = 2000; // 2 seconds
                const step = target / (duration / 16); // 60fps
                let current = 0;
                
                const timer = setInterval(() => {
                    current += step;
                    if (current >= target) {
                        current = target;
                        clearInterval(timer);
                    }
                    stat.textContent = Math.floor(current);
                }, 16);
            });
        }

        // 当数字进入视口时触发动画
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    animateNumbers();
                    observer.unobserve(entry.target);
                }
            });
        }, { threshold: 0.5 });

        observer.observe(document.querySelector('.stats'));
